void generareMidsquare()
{
    int seed;
	for (int i=1; i<= 30; i++)
	{
		seed = seed*(seed+19);
		seed = ((seed / 100)+96)%10000;
		cout << seed<<" ";
	}

}

float evaluare(int alegere)
{
    int x,y,i,j;

        switch(alegere)
